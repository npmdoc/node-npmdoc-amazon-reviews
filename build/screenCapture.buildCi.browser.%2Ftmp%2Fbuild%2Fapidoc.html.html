function amazon-reviews.getReview (options, callback)

    getReview = function (options, callback) {
  var reviewMobilePage;
  if ((options != null ? options.productId : void 0) == null) {
    return callback(new Error('no options.productId'));
  }
  if ((options != null ? options.reviewId : void 0) == null) {
    return callback(new Error('no options.reviewId'));
  }
  options.url = "http://www.amazon.com/gp/aw/review/" + options.productId + "/" + options.reviewId;
  return reviewMobilePage = new ReviewMobilePage(options, function(err, $) {
    var review, reviewDesktopPage;
    if (err != null) {
      return callback(err);
    }
    reviewMobilePage.$ = $;
    review = reviewMobilePage.parse();
    options.url = "http://www.amazon.com/review/" + options.reviewId;
    return reviewDesktopPage = new ReviewDesktopPage(options, function(err, $) {
      var k, v, _ref;
      if (err != null) {
        return callback(err);
      }
      reviewDesktopPage.$ = $;
      _ref = reviewDesktopPage.parse();
      for (k in _ref) {
        v = _ref[k];
        review[k] = v;
      }
      return callback(null, review);
    });
  });
}

    getReviewIds = function (options, callback) {
  var allCustomerReviewsPage, currentUrl, isLastPage, reviewIds;
  if (options.productId == null) {
    return callback(new Error('no options.productId'));
  }
  reviewIds = [];
  isLastPage = false;
  currentUrl = "http://www.amazon.com/product-reviews/" + options.productId + "/ref=cm_cr_pr_top_recent?ie=UTF8&showViewpoints=0
&sortBy=bySubmissionDateDescending";
  return allCustomerReviewsPage = new AllCustomerReviewsPage({
    url: currentUrl
  }, function(err, $) {
    var lastPageNo, nextPageUrl, reviewPageNos, _i, _results;
    if (err != null) {
      return callback(err);
    }
    allCustomerReviewsPage.$ = $;
    reviewIds = reviewIds.concat(allCustomerReviewsPage.getReviewIds());
    nextPageUrl = allCustomerReviewsPage.getNextPageUrl();
    lastPageNo = allCustomerReviewsPage.getLastPageNo();
    if (lastPageNo <= 1) {
      return callback(null, reviewIds);
    }
    reviewPageNos = (function() {
      _results = [];
      for (var _i = 2; 2 <= lastPageNo ? _i <= lastPageNo : _i >= lastPageNo; 2 <= lastPageNo ? _i++ : _i--){ _results.push(_i); }
      return _results;
    }).apply(this);
    return async.forEachLimit(reviewPageNos, 100, function(reviewPageNo, callback) {
      return allCustomerReviewsPage = new AllCustomerReviewsPage({
        url: nextPageUrl.replace(/pageNumber=[\d]*/g, "pageNumber=" + reviewPageNo)
      }, function(err, $) {
        if (err != null) {
          return callback(err);
        }
        allCustomerReviewsPage.$ = $;
        reviewIds = reviewIds.concat(allCustomerReviewsPage.getReviewIds());
        return callback(null);
      });
    }, function(err) {
      if (err != null) {
        return callback(err);
      }
      return callback(null, reviewIds);
    });
  });
}
